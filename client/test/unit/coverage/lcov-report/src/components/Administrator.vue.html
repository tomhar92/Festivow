<template>
  <div class="Administrator">
      <h1>Administrator Panel</h1>
        <v-btn
          color="teal"
          @click="presentUsers">
          Present Users
        </v-btn>
        <article v-for="(user, idx) in users" :key="idx">
          <h2>User Name: {{ user.name }}</h2>
          <h3>User Email: {{ user.email }}</h3>
          <v-btn
            color="teal" 
            @click="deleteUser(user.id)">
            Delete User
          </v-btn>
        </article>
      <br>    
  </div>
</template>
 
<script>
const firebase = require('../firebaseConfig.js')
export default {
  name: 'Administrator',
  data () {
    return {
      users: []
    }
  },
  methods: {
    presentUsers: function(){
      console.log('Trying to Show Users!')
      let usersFromDB = []
      firebase.usersCollection.get().then(function(querySnapshot){
        querySnapshot.forEach( function(doc) {
          let currentUser = {
            id: doc.id,
            name: doc.data().name,
            email: doc.data().email,
            country: doc.data().country,
            city: doc.data().city,
            age: doc.data().age
          }
          usersFromDB.push(currentUser)
          console.log("Added User " + doc.id + " To Users")
          // statements
        })
      })
      this.users = usersFromDB
    },
    
    deleteUser: function(id){
      console.log("Trying to delete " +id)
      firebase.usersCollection.doc(id).delete().then(function(){
       alert("Deleted Document Successfully")
      }).catch(function(error){
        alert(error)
      })
    }
  } 
}
</script>
