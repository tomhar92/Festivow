<template>
  
  <div class="profile" ref="container">
    <h1>User Profile</h1>
    <br>
    <v-layout justify-center>
      <v-responsive max-width="500px">
        <div class="details" v-if="exists">
          <v-layout row>
            <v-flex xs6>
              <h2>User Name</h2>
            </v-flex>
            <v-flex xs6>
              <h2>{{ user.name }}</h2>
            </v-flex>   
          </v-layout>
          <v-layout row>
            <v-flex xs6>
              <h4>E-Mail</h4>
            </v-flex>
            <v-flex xs6>
              <h4>{{user.email}}</h4>
            </v-flex>
          </v-layout>
          <v-layout row>
            <v-flex xs6>
              <h4>Age</h4>
            </v-flex>
            <v-flex xs6>
              <h4>{{user.age}}</h4>
            </v-flex>  
          </v-layout>
          <v-layout row>
            <v-flex xs6>
              <h4>City</h4>
            </v-flex>
            <v-flex xs6>
              <h4>{{user.city}}</h4>
            </v-flex>
          </v-layout>
          <v-layout row>
            <v-flex xs6>
              <h4>Country</h4>
            </v-flex>
            <v-flex xs6>
              <h4>{{user.country}}</h4>
            </v-flex>  
          </v-layout>
          <br>
          <v-layout row>
            <v-btn color="primary" @click="openPasswordDialog">Change Password</v-btn>
            <v-btn color="primary">Delete User</v-btn>
            <v-btn color="primary">Update Profile</v-btn>
          </v-layout>
        </div>
      </v-responsive>
    </v-layout>       
  </div>
</template>
 
<script>
import Vue from 'vue'
import passwordDialog from '@/components/changePasswordDialog.vue'
const firebase = require('../firebaseConfig.js')
 
export default {
  name: 'Administrator',
  data: function(){
    return{
      exists: false,
      
      user: {}
    }
  },
  created() {
        console.log("Trying to present user")
        let docId = firebase.currentUser.uid
        console.log("Looking for User : " +docId)
        firebase.usersCollection.doc(docId).get().then((doc) => {
          if(doc.exists){
            console.log(doc.data())
              this.exists = true
              this.user.id = doc.id
              this.user.name = doc.data().name
              this.user.email = doc.data().email
              this.user.country = doc.data().country
              this.user.city = doc.data().city
              this.user.age = doc.data().age
              this.user.reviews = doc.data().reviews
          }  
          else{
            console.error("Doc " +docId+ " dont exist!")
          }
        }).catch(function(error){
          console.error(error)
        })
      },
  methods: {
    openPasswordDialog: function(){
      var ComponentClass = Vue.extend(passwordDialog)
      var instance = new ComponentClass()
      instance.$mount()
      this.$refs.container.appendChild(instance.$el)
    }
  }    
  }     
</script>
