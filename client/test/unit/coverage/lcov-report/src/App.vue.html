<template>
  <v-app id="app">
  	  <v-toolbar color="cyan">
		<v-toolbar-title>Festivow</v-toolbar-title>
		<v-spacer></v-spacer>
			<v-toolbar-items v-if="logged">
				<v-btn flat @click="hello">Home Page</v-btn>
				<v-btn flat @click="profile">Profile</v-btn>
			</v-toolbar-items>
			<v-toolbar-items v-else>
				<v-btn flat @click="login">Login</v-btn flat>
			</v-toolbar-items>		
  	  </v-toolbar>	 
      <router-view/>
  </v-app>
</template>
 
<script>
import router from '../src/router/index.js'
const firebase = require('./firebaseConfig')
 
export default {
  name: 'App',
  computed: {
  	logged: function(){
        if(!firebase.currentUser){
        	return false
        }
        return true    	
  	}
  },
  methods: {
  	hello: function(){
  		router.push('hello')
  	},
  	profile: function(){
  		router.push('profile')
  	},
  	login: function(){
  		router.push('login')
  	}
  }	
}
</script>
